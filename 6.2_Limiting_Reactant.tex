\subsection*{Question 1}
$2.50~g$ of \ch{H2} and $18.2~g$ of \ch{O2} react according to the equation: \ch{2 H2(g) + O2(g) -> 2 H2O(g)}

\noindent
\begin{itemize}
	\item Which reactant is the \emph{limiting reactant}
	
	\vspace{3em}
	\item How many $g$ of water are produced?
	
	\vspace{3em}
	\item How many $g$ of the excess reactant remain?
	
	\vspace{3em}
	\item If $15.0~g$ of water are actually recovered, what is the \% yield?
\end{itemize}

\vspace{3em}
\subsection*{Question 2}
$5.00~g$ of \ch{CH4} and $20.0~g$ of \ch{O2} react according to the equation: \ch{CH4 (g) + 2 O2(g) -> CO2(g) + 2 H2O(g)}

\noindent
\begin{itemize}
	\item Which reactant is the \emph{limiting reactant}
	
	\vspace{3em}
	\item How many $g$ of water and carbon dioxide are produced?
	
	\vspace{6em}
	\item How many $g$ of the excess reactant remain?
	
	\vspace{3em}
	\item If $10.5~g$ of water are actually recovered, what is the \% yield?
\end{itemize}
